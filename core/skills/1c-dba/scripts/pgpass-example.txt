# Конфигурация скрипта обслуживания pg_maintenance.sh
# Нужно создать и настроить файл ~/.pgpass с правильными правами (0600) для пользователя, от имени которого будет запускаться этот скрипт.

# --- Параметры подключения к БД ---
# Хост БД (по умолчанию: localhost)
#PGHOST=your_db_host
# Порт БД (по умолчанию: 5432)
#PGPORT=5432
# Имя пользователя БД (ОБЯЗАТЕЛЬНО)
# Этот пользователь должен иметь достаточные права для выполнения VACUUM, ANALYZE, REINDEX,
# и, если используется pg_repack, права суперпользователя или членство в роли pg_repack.
PGUSER=postgres
# Имя базы данных 1С (ОБЯЗАТЕЛЬНО)
PGDATABASE=your_1c_database_name

# --- Аутентификация ---
# ВАЖНО: Для безопасности и удобства, пароль пользователя PGUSER
# должен быть указан в файле ~/.pgpass пользователя ОС, от имени которого
# будет запускаться этот скрипт (pg_maintenance.sh).
# Переменная окружения PGPASSWORD НЕ РЕКОМЕНДУЕТСЯ к использованию.
# См. подробные инструкции по настройке ~/.pgpass в шапке скрипта pg_maintenance.sh.
# Пример строки для ~/.pgpass:
# localhost:5432:your_1c_database_name:postgres:ваш_очень_секретный_пароль

# --- Настройки VACUUM ---
# Количество параллельных процессов для vacuumdb
VACUUMDB_JOBS=4

# --- Настройки REINDEX CONCURRENTLY ---
# Включить шаг переиндексации распухших индексов (true/false)
ENABLE_REINDEX=true
# Минимальный размер индекса в МБ для проверки на распухание
MIN_INDEX_SIZE_MB=100
# Порог распухания индекса в % для запуска REINDEX CONCURRENTLY
BLOAT_THRESHOLD_PERCENT=30

# --- Настройки PG_REPACK (Опционально) ---
# Включить шаг перепаковки распухших таблиц (true/false)
# Требует установленного расширения pg_repack в БД и утилиты на сервере
ENABLE_PG_REPACK=false
# Путь к исполняемому файлу pg_repack (если не в стандартном PATH)
#PG_REPACK_PATH=/opt/pg_repack/bin/pg_repack
# Минимальный размер таблицы в МБ для проверки на распухание
REPACK_MIN_TABLE_SIZE_MB=1024
# Порог распухания таблицы в % для запуска pg_repack
REPACK_BLOAT_THRESHOLD_PERCENT=25
# Количество параллельных задач для pg_repack (зависит от версии pg_repack)
REPACK_JOBS=2
